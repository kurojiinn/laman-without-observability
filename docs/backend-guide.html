Обзор проекта

Laman — стартап универсальной доставки в Грозном (400к население). Бэкенд написан на Go, обрабатывает заказы, авторизацию, трекинг курьеров и сборку товаров.

Кто пользуется системой

📱

CLIENT

Покупатель. Делает заказы через iOS или веб. Авторизуется по SMS OTP.

🛵

COURIER

Курьер. Доставляет заказы. Отправляет геолокацию, ведёт смены.

📦

PICKER

Сборщик в магазине. Собирает заказы, меняет статусы, добавляет замены.

🔧

ADMIN

Администратор. Управляет магазинами, товарами, видит статистику.

Dependency Injection — как всё связывается

В main.go зависимости создаются снизу вверх и передаются в конструкторы. Это позволяет легко менять реализации.

// Порядок инициализации в main.go:

// 1. Инфраструктура
db := database.New(cfg.Database)
redis := cache.New(cfg.Redis)

// 2. Репозитории (получают db/redis)
authRepo := auth.NewPostgresRepository(db)
orderRepo := orders.NewPostgresRepository(db)
courierRepo := courier.NewRedisRepository(redis)

// 3. Сервисы (получают репозитории)
authService := auth.NewService(authRepo, jwtSecret, smsProvider, otpLimiter)
orderService := orders.NewService(orderRepo, catalogService, courierService, ...)

// 4. Хэндлеры (получают сервисы)
authHandler := auth.NewHandler(authService)
orderHandler := orders.NewHandler(orderService)

// 5. Роутер (регистрирует хэндлеры)
router.RegisterRoutes(authHandler, orderHandler, ...)

Graceful Shutdown

// При получении Ctrl+C или kill сигнала:
quit := make(chan os.Signal, 1)
signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
<-quit // ждём сигнал

ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
defer cancel()

// Даём серверу 5 секунд завершить текущие запросы
srv.Shutdown(ctx)
db.Close()
redis.Close()

Это важно: без graceful shutdown запросы могут оборваться на середине. 5 секунд — время чтобы завершить текущие соединения.

auth — авторизация

OTP авторизация через SMS (SMS.ru API) + JWT токены для всех ролей. Для пикеров — дополнительно логин по паролю.

Как работает OTP

Пользователь вводит телефон
→
POST /request-code
→
4-знач. код сохраняется в БД (5 мин TTL)
→
SMS через SMS.ru
→
POST /verify-code + код
→
JWT токен

Rate Limiting через Redis

// Redis atomically считает попытки:
key := fmt.Sprintf("otp:attempts:%s", phone)
count, _ := redis.INCR(key)       // атомарный инкремент
redis.EXPIRE(key, 15*time.Minute) // сброс через 15 минут

if count > 5 {
    return ErrOTPBlocked // sentinel error
}

JWT токен

// Payload содержит только user_id
claims := jwt.MapClaims{
    "user_id": user.ID,
    "exp":     time.Now().Add(30 * 24 * time.Hour).Unix(),
}

// Middleware достаёт user из токена и кладёт в context:
ctx.Set("userID", claims["user_id"])
ctx.Set("userRole", user.Role)

Sentinel Errors

Именованные ошибки позволяют проверять тип ошибки без парсинга строк:

var (
    ErrOTPBlocked       = errors.New("otp: too many attempts")
    ErrUserAlreadyExists = errors.New("user: already exists")
    ErrInvalidCode      = errors.New("auth: invalid code")
)

// В handler'е:
if errors.Is(err, auth.ErrOTPBlocked) {
    c.JSON(429, gin.H{"error": "too many attempts"})
}

Как выглядит типичный Repository метод

func (r *postgresRepository) GetProducts(ctx context.Context, filter ProductFilter) ([]Product, error) {
    query := `SELECT id, name, price, image_url, is_available
               FROM products WHERE 1=1`

    var args []interface{}
    argNum := 1

    // Динамически строим WHERE в зависимости от фильтров
    if filter.CategoryID != nil {
        query += fmt.Sprintf(" AND category_id = $%d", argNum)
        args = append(args, *filter.CategoryID)
        argNum++
    }
    if filter.Search != "" {
        query += fmt.Sprintf(" AND name ILIKE $%d", argNum)
        args = append(args, "%"+filter.Search+"%")
    }

    var products []Product
    err := r.db.SelectContext(ctx, &products, query, args...)
    return products, err
}

Самый сложный пакет. Здесь живёт основная бизнес-логика: создание заказов, расчёт стоимости, state machine статусов, интеграция с курьерами и SSE.

Что происходит при CreateOrder()

// orders/service.go — упрощённо:
func (s *Service) CreateOrder(ctx context.Context, req CreateOrderRequest) (*Order, error) {
    // 1. Валидация: либо user_id, либо гостевые данные
    if req.UserID == nil && req.GuestName == "" {
        return nil, ErrInvalidRequest
    }

    // 2. Загружаем товары и считаем сумму
    products, _ := s.catalog.GetProductsByIDs(ctx, req.ProductIDs)
    itemsTotal := calculateTotal(products, req.Items)
    serviceFee := itemsTotal * 0.05   // 5% сервисный сбор
    deliveryFee := 200.0            // фиксированная доставка
    finalTotal := itemsTotal + serviceFee + deliveryFee

    // 3. Сохраняем заказ в транзакции
    order, _ := s.repo.CreateOrder(ctx, Order{...})
    s.repo.CreateOrderItems(ctx, order.ID, items)

    // 4. Создаём записи оплаты и доставки
    s.payments.Create(ctx, order.ID, req.PaymentMethod, finalTotal)
    s.delivery.Create(ctx, order.ID, req.Address)

    // 5. Ищем ближайшего курьера (5км → 10км → Telegram)
    courier, _ := s.courier.FindNearest(ctx, store.Lat, store.Lng, 5000)

    // 6. Уведомляем через SSE и Telegram
    s.hub.Notify(order.StoreID, orderEvent)
    s.telegram.Notify(ctx, "Новый заказ #"+order.ID)

    return order, nil
}

picker — панель сборщика

Сборщик работает в конкретном магазине, видит только свои заказы, и получает real-time уведомления через SSE.

SSE (Server-Sent Events) — что это

SSE — это постоянное HTTP соединение, по которому сервер может отправлять события клиенту в любой момент. Используется вместо WebSockets когда поток данных только сервер → клиент.

// events/hub.go — in-memory pub/sub по store_id
type Hub struct {
    subscribers map[string][]chan string // store_id → каналы
    mu          sync.RWMutex
}

// Picker подписывается:
ch := hub.Subscribe(storeID)
defer hub.Unsubscribe(storeID, ch)

// Когда создаётся заказ, orders уведомляет всех пикеров магазина:
hub.Notify(storeID, `{"type":"new_order","order_id":"..."}`)

Ручные позиции (замены товаров)

Если товара нет в наличии, сборщик может добавить замену вручную — без product_id из БД. Это делает product_id в order_items nullable.

POST /api/v1/picker/orders/:id/items
{
    "product_name": "Молоко Простоквашино 1л", // замена вручную
    "price": 89.90,
    "quantity": 2
    // product_id отсутствует — это нормально!
}

courier — трекинг курьеров

Redis GEO — специализированная структура данных для геопространственных запросов. Позволяет хранить координаты и делать запросы "найди всё в радиусе X км от точки".

Redis GEO команды

GEOADD active_couriers 45.7754 43.3059 "courier-uuid-123"

// Найти курьеров в 5 км от точки:
GEORADIUS active_couriers 45.7754 43.3059 5 km ASC COUNT 1

// Удалить курьера (конец смены):

middleware — сквозная логика

Middleware — это функции которые обрабатывают каждый запрос до или после handler'а. В Gin они регистрируются через router.Use().

Цепочка middleware

// main.go — порядок важен:
router.Use(
    middleware.RequestIDMiddleware(),  // 1. Присваивает X-Request-ID
    middleware.LoggingMiddleware(log), // 2. Логирует запрос
    middleware.MetricsMiddleware(),    // 3. Prometheus счётчики
    middleware.CORSMiddleware(cfg),    // 4. CORS заголовки
    middleware.RecoveryMiddleware(log),// 5. Перехватывает panic
)

// Для защищённых роутов:
protected := router.Group("/api/v1")
protected.Use(middleware.AuthMiddleware(authService))

AuthMiddleware — как читать токен

// Поддерживает два способа передачи токена:
// 1. Заголовок: Authorization: Bearer <token>
// 2. Query param: ?token=<token> (для SSE — браузер не шлёт заголовки)

token := c.GetHeader("Authorization")
if token == "" {
    token = c.Query("token")
}
token = strings.TrimPrefix(token, "Bearer ")

// WithTx — запускает несколько операций в одной транзакции

err := db.WithTx(ctx, func(tx *sqlx.Tx) error {
    if err := createOrder(tx, order); err != nil {
        return err // автоматический ROLLBACK
    }
    if err := createPayment(tx, payment); err != nil {

    }

Паттерны и советы

Лучшие практики в этом коде

Repository Pattern

Весь SQL изолирован в repository. Service не знает про PostgreSQL — только про интерфейс. Это позволяет легко менять БД или писать моки для тестов.

Context propagation

context.Context передаётся в каждый метод первым аргументом. Это позволяет отменять запросы, передавать trace ID и управлять таймаутами.

Early return

Ошибки проверяются сразу и функция возвращается. Нет глубокой вложенности if-ов. Код читается сверху вниз как список шагов.

Graceful degradation

SMS провайдер, Telegram, Jaeger — если не настроены, используются заглушки. Приложение стартует даже без них.

Dependency Injection

Все зависимости передаются в конструктор, не через global state. Функция New(db, logger, ...) явно объявляет что нужно сервису.

Interface Segregation

Интерфейсы маленькие и определяются там, где используются (consumer-driven). CourierService нужен orders только для FindNearest — именно этот метод и в интерфейсе.

Sentinel errors

var ErrNotFound = errors.New(...) позволяет в handler'е делать errors.Is(err, ErrNotFound) и возвращать правильный HTTP статус без парсинга строк.

%w при оборачивании ошибок

fmt.Errorf("create order: %w", err) — сохраняет оригинальную ошибку внутри. errors.Is/As могут её найти через всю цепочку.

dist/ в picker-panel

Папка dist/ — старый статичный билд. Docker запускает npm run dev, но исходники не маунтятся. При изменениях нужно пересобирать образ.

Миграции вручную

Миграции НЕ запускаются автоматически при старте. Нужно вручную: cd Backend && make migrate-up

SSE и токен в query

Браузеры не могут добавить Authorization заголовок к EventSource. Поэтому SSE эндпоинты принимают токен как ?token=... в URL.

JWT Secret в env

JWT_SECRET должен быть сложным и никогда не коммититься. Если утечёт — все токены можно подделать.

Типичная структура файлов пакета

internal/example/
├── handler.go     ← HTTP: parse request → call service → write response
├── service.go     ← Business logic: validation, orchestration, rules
├── repository.go  ← Data access: SQL queries, Redis commands
├── dto.go         ← Request/Response structs (Data Transfer Objects)
└── errors.go      ← Sentinel errors: var ErrNotFound = errors.New(...)

Как читать handler

func (h *Handler) CreateOrder(c *gin.Context) {
    // 1. Достаём данные из запроса
    var req CreateOrderRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }

    // 2. Достаём контекст пользователя (из middleware)
    userID, _ := c.Get("userID")

    // 3. Вызываем сервис (handler не знает про SQL)
    order, err := h.service.CreateOrder(c.Request.Context(), req)
    if err != nil {
        // 4. Конвертируем бизнес-ошибку в HTTP статус
        if errors.Is(err, ErrInvalidRequest) {
            c.JSON(400, gin.H{"error": err.Error()})
        } else {
            c.JSON(500, gin.H{"error": "internal error"})
        }
        return
    }

    // 5. Возвращаем результат
    c.JSON(201, order)
}
